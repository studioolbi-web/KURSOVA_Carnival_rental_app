-- Перевірка та створення таблиці категорій, якщо вона відсутня
-- Це забезпечує сумісність у разі розбіжностей у початкових міграціях

CREATE TABLE IF NOT EXISTS categories (
    id UUID PRIMARY KEY,
    name VARCHAR(255) NOT NULL UNIQUE,
    description TEXT
);

-- Також переконаємося, що в таблиці costumes є правильний зовнішній ключ
-- (H2 підтримує IF NOT EXISTS для багатьох операцій, але для ALTER TABLE краще бути обережним)
-- Оскільки Flyway гарантує послідовність, ми просто додаємо створення таблиці.
